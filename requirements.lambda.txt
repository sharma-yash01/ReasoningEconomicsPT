# Lambda Cloud GPU pinned stack for GRPO + TRL environment_factory + vLLM.
# GH200/aarch64: PyTorch skipped 2.8 for ARM64 CUDA builds; use torch 2.10.*.
# Keep vLLM within TRL 1.0.0's supported server range (0.10.2 through 0.17.1).
# All vllm versions >=0.11 require transformers<5. CARC (x86_64) uses vllm 0.10.2 which
# had no transformers upper bound and works with transformers==5.3.0. GH200 cannot use
# vllm 0.10.2 (no aarch64 CUDA torch 2.8 wheel), so transformers is pinned to <5 here.
# x86_64 (p5.4xlarge H100, CUDA 12.1): see requirements.carc-cu121.txt (torch 2.8.* is available there).
#
# Install (GH200 / CUDA 12.8):
#   pip install --upgrade pip
#   pip install -r requirements.lambda.txt \
#       --extra-index-url https://download.pytorch.org/whl/cu128
#
# With bootstrap_lambda.sh (GH200 / CUDA 12.8):
#   export REPT_REQUIREMENTS_FILE="$REPT_ROOT/requirements.lambda.txt"
#   export PYTORCH_WHEEL_INDEX="https://download.pytorch.org/whl/cu128"
#   export REPT_SKIP_TORCH_INSTALL=0   # must be 0 so bootstrap pre-installs CUDA torch
#   bash scripts/bootstrap_lambda.sh

# ---- Deep learning runtime ----
torch==2.10.*
# vllm 0.17.1 works with the NumPy 2.x stack used by the GH200/aarch64 wheels.
numpy>=2.0

# ---- Inference / generation ----
vllm==0.17.1
# vllm 0.17.1 pulls flashinfer-python 0.6.4; pin cubins too so downgrades
# from newer vLLM stacks do not leave a mismatched flashinfer-cubin package.
flashinfer-cubin==0.6.4

# ---- Core training stack ----
openenv-core[core]==0.2.*
trl==1.0.0
jmespath>=1.0.1
# transformers<5 required by vllm>=0.11; let pip resolve to latest 4.x compatible with trl
transformers>=4.56.0,<5
datasets>=4.7.0
accelerate>=1.4.0

# ---- Data / config ----
pydantic>=2.11.7
